Build the neural network#

[15]:

NUM_CLASSES = 3
NUM_FEATURES = 4

model = NN()

model.add_layer(Dense(NUM_FEATURES, 512, weight_regularizer_l2=5e-4, bias_regularizer_l2=5e-4))
model.add_layer(ReLU())
model.add_layer(Dropout(0.2))
model.add_layer(Dense(512, NUM_CLASSES))
model.add_layer(Softmax())

model.set_loss(CategoricalCrossentropy())
model.set_optimizer(Adam(learning_rate=0.005, decay=5e-5))
model.set_accuracy(MultiClassAccuracy())

model.build()
